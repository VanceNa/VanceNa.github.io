    问题2：Elements in iteration expect to have ‘v-bind:key’ directives.

    案例：

    <li v-for="menu in menus" @click="gotoThat(menu)" :class="{actColor:menu.selected}">
   {{menu.title}}
   <a :class="{active:menu.selected}"></a>
</li>

    解决方案：加:key

    <li v-for="(menu,index) in menus" :key="index" @click="gotoThat(menu)" :class="{actColor:menu.selected}">
        {{menu.title}}
        <a :class="{active:menu.selected}"></a>
</li>

    问题3：图片找不到

    案例：

     <template>
      <div>
          <img :src="msgInfo.imgUrl" />
      </div>
    </template>
    <script>
    export default {
      data () {
        return {
          msgInfo: {imgUrl: '../assets/logo.png'}
        }
      }
    }
    </script>

    解决方案：必须先导入图片 import logo from ‘../assets/logo.png’

     <template>
  <div>
      <img :src="msgInfo.imgUrl" />
  </div>
</template>
<script>
import logo from '../assets/logo.png'
export default {
  data () {
    return {
      msgInfo: {imgUrl: logo}
    }
  }
}
</script>

    问题5 ：不相关的两个组件之间如何传值

    解决方案：使用vuex中的store，所有的属性都必须放在state下面。在main.js中 需要将它挂载到vue上。

    store/index.js

    import Vue from 'vue'
import Vuex from 'vuex'
Vue.use(Vuex)
export default new Vuex.Store({
  state: {
    gameNum: 0
  }
})

    main.js

    import Vue from 'vue'
import App from './App'
import router from './router'
import axios from '@/utils/http.js'
import store from './store/index'
import './style/reset.css'

Vue.config.productionTip = false
Vue.prototype.axios = axios

new Vue({
  el: '#app',
  router,
  store,
  template: '<App/>',
  components: { App }
})

    问题6：props 静态类型检测

    案例：

    prop: [
  'id',
  'multiple'
]

    解决方案：

    props: {
  id: {
    type: [ Number, Array ],
    required: true,
  },
  multiple: {
    type: Boolean,
    default: false,
  }
}

    解决方案：
